        int playgroundSizeY;
    submodules:
        host[numHosts]: WirelessHostSimplified {
            @display("r=,,#707070");

            @display("p=213,174;r=,,#707070");

            playgroundSizeY = playgroundSizeY;
            numChannels = 2;

        configurator: FlatNetworkConfigurator {

            netmask = "255.255.0.0";
            @display("p=140,50");
        }
}
